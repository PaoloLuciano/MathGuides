\section{Matrices}
\begin{itemize}
	\item \textbf{Inversa de una Matriz 2x2}
	$$A = \begin{bmatrix}
	a&b\\c&d
	\end{bmatrix}
	\iff \det(A) = ad -cd \neq 0 \Rightarrow A^{-1} = \dfrac{1}{ad-cd}				\begin{bmatrix}
	d & -b \\
	-c & a
	\end{bmatrix}$$
\end{itemize}
